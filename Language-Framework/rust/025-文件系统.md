# 文件夹

```rust
use std::fs;
use std::path::Path;

fn main() {
  // 判断路径是否存在
  Path::new("some/path").exists();

  // 创建文件夹
  fs::create_dir("some_dir");

  // 递归创建文件夹
  fs::create_dir_all("/some/dir/what");

  // 删除空文件夹
  fs::remove_dir("some/dir");
  
  // 删除文件夹 (删除文件夹后所有文件后删除文件夹)
  fs::remove_dir_all("some/dir");
}




```
